<title>表格</title>

<!--表格结构-->

	<table>                         <!--定义表格-->
		<caption>表格标题</caption>
		<thead>                     <!--定义表格的页眉-->

		</thead>
		<tbody>                     <!--定义表格的主体-->
			<tr>                    <!--定义表格表头-->
				<th>行1, 头列1</td>
				<th>行1, 头列2</td>
			</tr>
			<tr>                    <!--定义表格行-->
				<td>行2, 列1</td>   <!--定义表格列-->
				<td>行2, 列2</td>
			</tr>
		</tbody>
		<tfoot>                     <!--定义表格的页脚-->

		</tfoot>
	</table>

<!--单元格合并-->

	colspan='n'                     向右横向合并单元格
	rowspan='n'                     向下纵向合并单元格
	border-collapse: collapse;      合并边框线

<!--表格边距-->

	<table>支持margin/padding
	<tr>, <td>, <th>支持padding, 不支持margin.
	可以通过给<table> border-spacing: X Y;(只给一个值X=Y)属性设置单元格之间的距离.

<!--表格特征-->

	表格默认内容撑开宽度
	<th>内容默认加粗, 水平垂直居中
	<td>内容默认水平居左, 垂直居中
	默认 内容均分表格水平空间

<!--来自菜鸟教程-->

	border='1/0'                    规定表格是否有边框
	cell-spacing='px';              规定单元格之间的空白。
	cell-padding='px';              规定单元边沿与其内容之间的空白。
